1. Agent Base Class
All agents inherit from this base class. Located in packages/core/agents/AgentBase.ts
Structure
export abstract class AgentBase {
 protected client: Anthropic
 protected tokenMeter: TokenMeter
 protected contextManager: ContextManager
 protected logger: Logger
 protected organizationId: UUID
 protected clientId: UUID
 protected campaignId: UUID

 constructor(deps: AgentDependencies)
 protected abstract getSystemPrompt(): string
 protected abstract parseResponse(response: Message): unknown
 protected async loadContext(): Promise<ContextBundle>
 protected async callClaude(userPrompt: string, context: ContextBundle): Promise<Message>
 protected async recordTokenUsage(response: Message, agentType: string)
 protected validateOutput(output: unknown): void
}
Example: Ideation Agent
export class IdeationAgent extends AgentBase {
 async execute(brief: ClientBrief): Promise<CampaignIdea[]> {
 const context = await this.loadContext()
 const systemPrompt = this.getSystemPrompt()
 const userPrompt = this.buildUserPrompt(brief, context)
 const response = await this.callClaude(userPrompt, context)
 await this.recordTokenUsage(response, 'ideation')
 const ideas = this.parseResponse(response) as CampaignIdea[]
 this.validateOutput(ideas)
 return ideas
 }
 protected getSystemPrompt(): string { ... }
 protected parseResponse(response: Message): unknown { ... }
}

export const campaignMachine = createMachine({
 id: 'campaign',
 initial: 'briefing',

 states: {
 briefing: {
 on: { BRIEF_SUBMITTED: { target: 'ideation', actions: 'setBrief' } }
 },

 },

 scoring: {
 invoke: { src: 'scoringAgent', onDone: { target: 'scoreReview', actions: 'setScore' } }
 },
 scoreReview: { type: 'final' }, // Publish or iterate
 }
 }, }
3. API Route Patterns
Campaign Creation Endpoint
// POST /api/campaigns
router.post('/campaigns', authenticate, async (req: AuthRequest, res) => {
 try {
 const { clientId, brief } = req.body
 validateUserAccess(req.user, clientId)
 const campaign = await campaignRepo.create({
 organizationId: req.org.id,
 clientId,
 brief,
 status: 'ideation_pending'
 })
 await contextManager.initializeCampaignContext(clientId, campaign.id, brief)
 res.status(201).json(campaign)
 } catch (err) {
 logger.error('Campaign creation failed', { error: err, userId: req.user.id })
 res.status(500).json({ error: 'Internal server error' })
 }
})
Ideation Endpoint
// POST /api/campaigns/:campaignId/ideate
router.post('/campaigns/:campaignId/ideate', authenticate, async (req, res) => {
 const campaign = await campaignRepo.findById(req.params.campaignId)
 validateUserAccess(req.user, campaign.clientId)
 const agent = new IdeationAgent({ client, tokenMeter, contextManager, ... })
 const ideas = await agent.execute(campaign.brief)
 await campaignRepo.update(campaign.id, { ideas, status: 'idea_review' })
 res.json({ ideas, campaignId: campaign.id })
})

5. Context File Operations
Context Manager Implementation
export class ContextManager {
 constructor(private storePath: string, private validator: SchemaValidator) {}

 async initializeCampaignContext(
 clientId: UUID,
 campaignId: UUID,
 brief: ClientBrief
 ): Promise<void> {
 const context: CampaignContext = {
 campaignId,
 clientId,
 createdAt: new Date().toISOString(),
 version: 1,
 campaignBrief: brief,
 previousCampaigns: await this.loadPreviousCampaigns(clientId),
 }
 this.validator.validate(context)
 const filePath = this.getContextFilePath(clientId, campaignId)
 await fs.promises.writeFile(filePath, JSON.stringify(context, null, 2))
 }

 async appendPerformanceAnalysis(
 campaignId: UUID,
 analysis: PerformanceAnalysis
 ): Promise<void> {
 const context = await this.loadCampaignContext(campaignId)
 context.performanceAnalysis = analysis
 context.lastUpdated = new Date().toISOString()
 const filePath = this.getContextFilePath(context.clientId, campaignId)
 await fs.promises.writeFile(filePath, JSON.stringify(context, null, 2))
 }
}

Prompt Injection Detection
export class PromptGuard {
 static detectInjection(input: string): boolean {
 const suspiciousPatterns = [
 /ignore.*previous.*instructions/i,
 /system.*prompt/i,
 /you are now/i,
 ]
 return suspiciousPatterns.some(pattern => pattern.test(input))
 }
}
Data Isolation Validation
// Middleware: Ensure user can only access their clients
export const validateClientAccess = (req: AuthRequest, res, next) => {
 const clientId = req.params.clientId || req.body.clientId
 if (!req.user.clientIds.includes(clientId)) {
 return res.status(403).json({ error: 'Access denied' })
 }
 next()
}
